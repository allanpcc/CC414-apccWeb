Crear una Tabla de DynamoDB con un template de CloudFormation

  resources:
    Resources:
      MarvelTable:
        Type: "AWS::DynamoDB::Table"
        Properties:
          AttributeDefinitions:
            -
              AttributeName: "Id"
              AttributeType: "S"

          KeySchema:
            -
              AttributeName: "Id"
              KeyType: "HASH"
          ProvisionedThroughput:
          ReadCapacityUnits: 10
          WriteCapacityUnits: 10

         TableName: Allan-Marvel-logs
